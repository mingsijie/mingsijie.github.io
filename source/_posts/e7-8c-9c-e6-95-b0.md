    #P121  4.1猜数游戏
    import random
    待猜数 = random.randrange(0,9)
    次数 = 0
    while True:
        猜数 = int(input("请猜数： "))
        次数 += 1
        if 猜数 == 待猜数:
            print("预测{}次，你猜中了！".format(次数))
            break
        elif 猜数<= 待猜数 :
            print("遗憾，太小了")
        else :
            print("遗憾，太大了")
